--In ORACLE the option "ON DELETE RESTRICT" or "ON DELETE NO ACTION" are implemented by default and doesn't have specific syntax. So we just add ON DELETE CASCADE when it is necessary. Also, in Oracle, there is no ON UPDATE CASCADE option. We would have to make a trigger for each case if we wanted to.
-- ASSOCIATION -> DISCIPLINES 
ALTER TABLE ASSOCIATION 
ADD CONSTRAINT fk_association_discipline 
FOREIGN KEY (discipline_code) 
REFERENCES DISCIPLINES(code);

ALTER TABLE ASSOCIATION 
ADD CONSTRAINT fk_association_community 
FOREIGN KEY (community) 
REFERENCES MEMBERS(code);

-- GROUP_DISCIPLINE -> RESEARCH_GROUP and DISCIPLINES
ALTER TABLE GROUP_DISCIPLINE 
ADD CONSTRAINT fk_group_discipline_group 
FOREIGN KEY (group_code) 
REFERENCES RESEARCH_GROUP(code);

ALTER TABLE GROUP_DISCIPLINE 
ADD CONSTRAINT fk_group_discipline_discipline 
FOREIGN KEY (discipline_code) 
REFERENCES DISCIPLINES(code);

-- MEMBERS -> RESEARCH_GROUP & CATEGORY
ALTER TABLE MEMBERS 
ADD CONSTRAINT fk_members_group 
FOREIGN KEY (research_group_code) 
REFERENCES RESEARCH_GROUP(code);

ALTER TABLE MEMBERS 
ADD CONSTRAINT fk_members_category 
FOREIGN KEY (code_category) 
REFERENCES CATEGORY(code);

-- MEMBERS_ASSOCIATION -> ASSOCIATION & MEMBERS
ALTER TABLE MEMBERS_ASSOCIATION 
ADD CONSTRAINT fk_ma_association 
FOREIGN KEY (association_name) 
REFERENCES ASSOCIATION(name);

ALTER TABLE MEMBERS_ASSOCIATION 
ADD CONSTRAINT fk_ma_member 
FOREIGN KEY (member_code) 
REFERENCES MEMBERS(code);

-- BOARD_ASS & BOARD_CONF -> BOARD & MEMBERS
ALTER TABLE BOARD_ASS 
ADD CONSTRAINT fk_boardass_ma 
FOREIGN KEY (name_association, code_member) 
REFERENCES MEMBERS_ASSOCIATION(association_name, member_code);

-- BOARD_ASS -> BOARD
ALTER TABLE BOARD_ASS 
ADD CONSTRAINT fk_code_board 
FOREIGN KEY (code_board) 
REFERENCES BOARD(code);

ALTER TABLE BOARD_CONF 
ADD CONSTRAINT fk_boardconf_board 
FOREIGN KEY (code_board) 
REFERENCES BOARD(code);

ALTER TABLE BOARD_CONF 
ADD CONSTRAINT fk_boardconf_member 
FOREIGN KEY (code_member) 
REFERENCES MEMBERS(code);

-- MEMBERS_FEE -> MEMBERS & FEE
ALTER TABLE MEMBERS_FEE 
ADD CONSTRAINT fk_mf_member 
FOREIGN KEY (code_member) 
REFERENCES MEMBERS(code);

ALTER TABLE MEMBERS_FEE 
ADD CONSTRAINT fk_mf_fee 
FOREIGN KEY (code_fee) 
REFERENCES FEE(code);

-- PRIZE -> MEMBERS
ALTER TABLE PRIZE 
ADD CONSTRAINT fk_prize_member 
FOREIGN KEY (code_member) 
REFERENCES MEMBERS(code);

-- NETWORKS_CONFEDERATION & NETWORKS_ASSOCIATION -> SOCIAL_NETWORK & MEMBERS
ALTER TABLE NETWORKS_CONFEDERATION 
ADD CONSTRAINT fk_nc_network 
FOREIGN KEY (code_social_network) REFERENCES SOCIAL_NETWORK (CODE) ON DELETE CASCADE;

ALTER TABLE NETWORKS_CONFEDERATION  
ADD CONSTRAINT fk_code_community
FOREIGN KEY (community) 
REFERENCES MEMBERS(code);

ALTER TABLE NETWORKS_ASSOCIATION 
ADD CONSTRAINT fk_na_network 
FOREIGN KEY (code_social_network) REFERENCES SOCIAL_NETWORK (CODE) ON DELETE CASCADE;

ALTER TABLE NETWORKS_ASSOCIATION 
ADD CONSTRAINT fk_na_association 
FOREIGN KEY (association_name) 
REFERENCES ASSOCIATION(name);
